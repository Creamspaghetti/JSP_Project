-- DB 만들기
-- Table: Users
CREATE TABLE Users (
    USER_ID 		VARCHAR2(300) PRIMARY KEY,
    user_pw 		VARCHAR2(300),
    user_name 		VARCHAR2(300),
    user_phone 		VARCHAR2(300),	-- 숫자로 바꿀까?
    user_gender 	VARCHAR2(300),
    user_address 	VARCHAR2(300)
);

-- Table: Products
CREATE TABLE Products (
    PRODUCT_ID 			VARCHAR2(300) PRIMARY KEY,
    Brand 				VARCHAR2(300),
    Product_name 		VARCHAR2(300),
    Price_amount 		NUMBER,    -- 숫자임
    normal_price 		NUMBER,    -- 숫자임
    sale_rate 			NUMBER,       -- 숫자임
    Image_url 			VARCHAR2(300),
    CATEGORY 			varchar2(200),
    subCategory			varchar2(200)	
);

-- Table: Cart
CREATE TABLE Cart (
    CART_NUM 	VARCHAR2(300) PRIMARY KEY,
    USER_ID 	VARCHAR2(300),
    PRODUCT_ID 	VARCHAR2(300),
    FOREIGN KEY (USER_ID) REFERENCES Users(USER_ID),
    FOREIGN KEY (PRODUCT_ID) REFERENCES Products(PRODUCT_ID)
);

-- Table: Order_
CREATE TABLE Order_ (
    ORDER_NUM 	VARCHAR2(300) PRIMARY KEY,
    PRODUCT_ID 	VARCHAR2(300),
    FOREIGN KEY (PRODUCT_ID) REFERENCES Products(PRODUCT_ID)
);

-- Table: Ordered
CREATE TABLE Ordered (
    ORDERED_NUM 	VARCHAR2(300) PRIMARY KEY,
    ORDER_NUM 		VARCHAR2(300),
    ordered_date 	DATE,
    FOREIGN KEY (ORDER_NUM) REFERENCES Order_(ORDER_NUM)
);

-- Table: Product_count
-- 이건 나중에 시간 나면 연결할거임
CREATE TABLE Product_count (
    Product_count NUMBER PRIMARY KEY  -- 숫자임
);
